{
    let countA = 0;
    let countD = 0;

      if(S.charAt(i) === "A"){
        countA++;

        countD++;
      }
    }
    let arr = [countA, countD];
    return arr;
};

var Count_Occ = (s) => 
{
  let n = s.length;
  let str = s.split("").sort().join("");
  let ans = "";
  
  for(let i = 0; i < n; i++){
    let count = 0;
    for(let j = 0; j < n; j++){
      if(str[i] === str[j] && i > j){
        break;
      }
      if(str[i] === str[j]){
        count++;
      }
    }
    if(count > 1){
      ans += str[i] + count;
    }
  }
  return ans;
};

var Game_Winner = (S) => 
{
  let countA = 0, countB = 0;
  for(let i = 0; i < S.length; i++){
    if(S.charAt(i) === "A"){
      countA++;
    }
    else if(S.charAt(i) === "D"){
      countB++;
    }
  }
  if(countA === countB){
    return "Draw";
  }
  else if(countA > countB){
    return "Aditya";
  }
  else {
    return "Danish";
  }
}
